    function DatePickerWithRestrictionsExample() {
      const [date, setDate] = React.useState<Date>();

      const disableWeekends = (date: Date) => {
        return date.getDay() === 0 || date.getDay() === 6;
      };

      return (
        <div className="space-y-4 w-full max-w-sm">
          <DatePicker
            value={date}
            onChange={setDate}
            placeholder="Future dates only"
            minDate={new Date()}
          />

            value={date}
            onChange={setDate}
            placeholder="Weekdays only"

### Booking System

Great for hotel bookings, appointment scheduling, and reservation systems.

    ```tsx
    function BookingDatePickerExample() {
      const [checkIn, setCheckIn] = React.useState<Date>();
      const [checkOut, setCheckOut] = React.useState<Date>();

      const bookedDates = [
        new Date(2025, 5, 15),
        new Date(2025, 5, 16),
        new Date(2025, 5, 20),
        new Date(2025, 5, 25),
      ];

      const isDateBooked = (date: Date) => {
        return bookedDates.some(
          (bookedDate) => date.toDateString() === bookedDate.toDateString()
        );
      };

      const disableUnavailableDates = (date: Date) => {
        const today = new Date();
        today.setHours(0, 0, 0, 0);
        return date < today || isDateBooked(date);
      };

      return (
        <div className="space-y-3">
          <DatePicker
            value={checkIn}
            onChange={setCheckIn}
            placeholder="Check-in date"
            disabledDates={disableUnavailableDates}
            minDate={new Date()}
          />
          <DatePicker
            value={checkOut}
            onChange={setCheckOut}
            placeholder="Check-out date"
            disabledDates={disableUnavailableDates}
            minDate={checkIn ? new Date(checkIn.getTime() + 24 * 60 * 60 * 1000) : new Date()}
          />
        </div>
      );
    }
    ```

  </Tab>

## Features

- **Single & Range Selection**: Support for both single date and date range selection
- **Keyboard Navigation**: Full keyboard support with arrow keys and Enter/Space
- **Accessibility**: ARIA labels, focus management, and screen reader support
- **Date Restrictions**: Min/max dates and custom disabled date functions
- **Custom Formatting**: Flexible date formatting with locale support
- **Responsive Design**: Mobile-optimized with touch-friendly interactions
- **Body Scroll Lock**: Prevents page scrolling when calendar is open
- **Animation**: Smooth open/close animations with Framer Motion
- **Variants & Sizes**: Multiple visual variants and size options
- **Form Integration**: Easy integration with form libraries
